80 template <typename T> static constexpr T reflect(const T &x, const size_t &M) {
81 
82  static_assert(std::is_arithmetic<T>::value, "Not an arithmetic type");
83 
84  const size_t m = std::min(M, 8 * sizeof(T));
85 
86  T y = 0;
87 
88  for (size_t i = 0; i < m; ++i) {
89  bool bit = (x >> i) & 1;
90  y |= bit << ((m - 1) - i);
91  }
92 
93  return y;
94 }

135 template <typename T, std::size_t N, std::uintmax_t polynomial>
136 struct generator {
137 
139  static constexpr void checkTemplateParameters() {
140  static_assert(std::is_arithmetic<T>::value, "Not an arithmetic type");
141  static_assert(0 != N, "CRC length must be non-zero");
142  static_assert(0 == N % 8, "CRC length must be a multiple of 8");
143  static_assert(N <= 8 * sizeof(T), "CRC length must be <= 8 * sizeof(T)");
144  static_assert(T(0) != polynomial, "CRC polynomial must be non-zero");
145  static_assert(T(0) == (polynomial & (~mask())),
146  "CRC polynomial must be N bits");
147  }
148 
150  static constexpr T msb() {
151  static_assert(N <= 8 * sizeof(T), "CRC length must be <= 8 * sizeof(T)");
152  return T(1ULL << (N - 1));
153  }
154 
156  static constexpr T mask() {
157  static_assert(N <= 8 * sizeof(T), "CRC length must be <= 8 * sizeof(T)");
158  if (N == 8 * sizeof(T)) {
159  return T(-1);
160  } else {
161  return T((1ULL << N) - 1);
162  }
163  }
164 
166  static constexpr T func(T x) {
167 
168  checkTemplateParameters();
169 
170  T index = (x << (N - 8)) & mask();
171  for (auto bit = 0; bit < 8; ++bit) {
172  if (0 != (index & msb())) {
173  index <<= 1;
174  index ^= polynomial;
175  } else {
176  index <<= 1;
177  }
178  }
179 
180  return index & mask();
181  }
182 };
